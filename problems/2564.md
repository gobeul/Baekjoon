W, H = map(int, input().split())

# 북 1[0] // 남 2[1] // 서 3[2] // 동 4[3]

square = [[False]*W, [False]*W, [False]*H, [False]*H]

N = int(input())
for i in range(N):
    store_direction, store_distance = map(int, input().split())
    # 상점 표시하기
    square[store_direction-1][store_distance -1]  = True

my_direction, my_distance = map(int, input().split())
square[my_direction-1][my_distance-1] = 2

ans = 0
if my_direction == 1: # 내가 북쪽에 있는 경우 서 - 북 - 동
    my_rest = [my_distance, W-my_distance]
    line = square[2][::-1]
    line.extend(square[0])
    line.extend(square[3])
    # 맞은편 방위 거리 계산.
    rest = []
    for i, v in enumerate(square[1]):
        if v:
            rest.append(i+1) # 기준 점으로 부터 방향 거리 (인덱스 +1) 
    for i in range(len(rest)):
        rest[i] = [rest[i], W-rest[i]]  # [기준점으로부터의 거리, 기준점 반대로 부터의 거리]

    for i in rest:
        a = i[0] + my_rest[0]
        b = i[1] + my_rest[1]
        ans += min(a, b) + H

elif my_direction == 2: # 남쪽 서 - 남 - 동
    my_rest = [my_distance, W-my_distance]
    line = square[2]
    line.extend(square[1])
    line.extend(square[3][::-1])
    print(line)
    rest = []
    for i, v in enumerate(square[0]):
        if v:
            rest.append(i+1)
    for i in range(len(rest)):
        rest[i] = [rest[i], W-rest[i]]
    
    for i in rest:
        a = i[0] + my_rest[0]
        b = i[1] + my_rest[1]
        ans += min(a, b) + H

elif my_direction == 3: # 서쪽 남 - 서 - 북
    my_rest = [my_distance, H-my_distance]
    line = square[1][::-1]
    line.extend(square[2][::-1])
    line.extend(square[0])
    rest = []
    for i, v in enumerate(square[3]):
        if v:
            rest.append(i+1)
    
    for i in range(len(rest)):
        rest[i] = [rest[i], H-rest[i]]
    
    for i in rest:
        a = i[0] + my_rest[0]
        b = i[1] + my_rest[1]
        ans += min(a, b) + W


else:                   # 동쪽 북 - 동 - 남
    my_rest = [my_distance, H-my_distance]
    line = square[0]
    line.extend(square[3])
    line.extend(square[1][::-1])
    rest = []
    for i, v in enumerate(square[2]):
        if v:
            rest.append(i+1)
    
    for i in range(len(rest)):
        rest[i] = [rest[i], H-rest[i]]

    for i in rest:
        a = i[0] + my_rest[0]
        b = i[1] + my_rest[1]
        ans += min(a, b) + W


my_point = line.index(2) # 일자 라인에서 내 위치 Treu == 1


for i, v in enumerate(line):
    if v:
        ans += abs(i-my_point)


print(ans)
